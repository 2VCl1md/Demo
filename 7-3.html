function quicksort(arr){
if (arr.length<=1) {
return arr;
}
var left=[];
var right=[];
var mid=Math.floor(arr.length/2);
var val=arr.splice(mid,1)
for (var i = 0; i < arr.length; i++) {
if (arr[i]>val) {
right.push(arr[i])
}
else{
left.push(arr[i])
}
}
return quicksort(left).concat([val],quicksort(right));
}
console.log(quicksort([21,555,31,1,848,3,44,6]));
